تمرین ۷ متوسط

تمرین ۸ متوسط
تابعی به اسم calculate بنویس که دو عدد و یه عملگر (+, -, *, /) بگیره و نتیجه رو برگردونه.
مشاهده جواب
def calculate(a, b, op):
    if op == "+":
        return a + b
    elif op == "-":
        return a - b
    elif op == "*":
        return a * b
    elif op == "/":
        if b != 0:
            return a / b
        else:
            return "خطا: تقسیم بر صفر"
    else:
        return "عملگر نامعتبر"

تمرین ۱۱ سخت
تابعی به اسم is_prime بنویس که یه عدد بگیره و تشخیص بده اول هست یا نه.
مشاهده جواب
def is_prime(n):
    if n < 2:
        return False
    for i in range(2, n):
        if n % i == 0:
            return False
    return True

print(is_prime(7))
print(is_prime(10))
خروجی
True
False

تمرین ۱۴ سخت
تابعی به اسم count_vowels بنویس که یه رشته بگیره و تعداد حروف صدادار (a, e, i, o, u) رو بشمره.
مشاهده جواب
def count_vowels(text):
    vowels = "aeiouAEIOU"
    count = 0
    for char in text:
        if char in vowels:
            count += 1
    return count

print(count_vowels("Hello World"))
خروجی
3
✨ پروژه نهایی توابع پروژه
یه برنامه ماشین حساب پیشرفته با توابع بنویس که:
• تابع show_menu() - منو رو نشون بده
• تابع add(a, b) - جمع
• تابع subtract(a, b) - تفریق
• تابع multiply(a, b) - ضرب
• تابع divide(a, b) - تقسیم (با مدیریت تقسیم بر صفر)
• تابع power(a, b) - توان
• تابع factorial(n) - فاکتوریل
• تابع main() - حلقه اصلی برنامه که منو رو نشون بده و از کاربر انتخاب بگیره و توابع مناسب رو صدا بزنه
مشاهده جواب
def show_menu():
    print("\n" + "="*40)
    print("ماشین حساب پیشرفته")
    print("="*40)
    print("1. جمع")
    print("2. تفریق")
    print("3. ضرب")
    print("4. تقسیم")
    print("5. توان")
    print("6. فاکتوریل")
    print("7. خروج")
    print("="*40)

def add(a, b): return a + b
def subtract(a, b): return a - b
def multiply(a, b): return a * b
def divide(a, b):
    if b == 0:
        return "خطا: تقسیم بر صفر"
    return a / b
def power(a, b):
    result = 1
    for i in range(b):
        result *= a
    return result
def factorial(n):
    if n < 0:
        return "خطا: فاکتوریل عدد منفی تعریف نشده"
    result = 1
    for i in range(1, n + 1):
        result *= i
    return result

def main():
    while True:
        show_menu()
        choice = input("انتخاب خود را وارد کنید: ")
        
        if choice == "7":
            print("خدانگهدار!")
            break
        
        if choice == "6":
            n = int(input("عدد را وارد کنید: "))
            print(f"فاکتوریل: {factorial(n)}")
            continue
        
        a = float(input("عدد اول: "))
        b = float(input("عدد دوم: "))
        
        if choice == "1":
            print(f"نتیجه: {add(a, b)}")
        elif choice == "2":
            print(f"نتیجه: {subtract(a, b)}")
        elif choice == "3":
            print(f"نتیجه: {multiply(a, b)}")
        elif choice == "4":
            print(f"نتیجه: {divide(a, b)}")
        elif choice == "5":
            print(f"نتیجه: {power(int(a), int(b))}")
        else:
            print("انتخاب نامعتبر!")

main()
