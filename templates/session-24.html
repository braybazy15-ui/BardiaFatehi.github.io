تمرین لیست‌ها

به جلسه بیست و چهارم خوش اومدی! توی سه جلسه قبل با لیست‌ها، متدها و پیمایش لیست آشنا شدیم. حالا وقتشه با تمرین‌های متنوع این مفاهیم رو کامل تثبیت کنیم.

این جلسه ۱۵ تمرین آماده کردم که تمام مباحث لیست رو پوشش میده. سعی کن اول خودت حل کنی، بعد جواب رو ببین.

بردیا فاتحی

ایجاد و دسترسی

my_list = [1, 2, 3]
print(my_list[0]) # 1

متدها

append(), insert(), remove()
pop(), sort(), reverse()

پیمایش

for item in list:
for i in range(len(list)):

برش

list[start:end:step]

تمرینات لیست‌ها (۱۵ تمرین)

تمرین ۱ آسان
یه لیست از ۵ رنگ مورد علاقت بساز و با حلقه for همه رو چاپ کن.
مشاهده جواب
colors = ["آبی", "قرمز", "سبز", "زرد", "بنفش"]
for color in colors:
    print(color)
تمرین ۲ آسان
لیست [10, 20, 30, 40, 50] رو بساز. اولین و آخرین عضو رو با ایندکس چاپ کن.
مشاهده جواب
nums = [10, 20, 30, 40, 50]
print("اولین:", nums[0])
print("آخرین:", nums[-1])
تمرین ۳ آسان
با استفاده از append، اعداد ۱ تا ۵ رو به یه لیست خالی اضافه کن.
مشاهده جواب
numbers = []
for i in range(1, 6):
    numbers.append(i)
print(numbers)
تمرین ۴ متوسط
لیست [5, 2, 8, 1, 9, 3] رو مرتب کن (صعودی) و بعد برعکسش کن.
مشاهده جواب
nums = [5, 2, 8, 1, 9, 3]
nums.sort()
print("مرتب:", nums)
nums.reverse()
print("برعکس:", nums)
تمرین ۵ متوسط
لیست [1, 2, 3, 4, 5] رو با برش به دو تا لیست تقسیم کن:
• سه عضو اول
• دو عضو آخر
مشاهده جواب
nums = [1, 2, 3, 4, 5]
first = nums[:3]
last = nums[3:]
print("سه عضو اول:", first)
print("دو عضو آخر:", last)
تمرین ۶ متوسط
توی لیست [12, 45, 23, 78, 34, 56] بزرگترین عدد رو پیدا کن (بدون استفاده از max).
مشاهده جواب
nums = [12, 45, 23, 78, 34, 56]
max_num = nums[0]
for num in nums:
    if num > max_num:
        max_num = num
print("بزرگترین:", max_num)
تمرین ۷ متوسط
یه لیست از اعداد ۱ تا ۱۰ بساز. مجموع اعداد رو حساب کن و میانگین رو چاپ کن.
مشاهده جواب
nums = list(range(1, 11))
total = 0
for num in nums:
    total += num
average = total / len(nums)
print("مجموع:", total)
print("میانگین:", average)
تمرین ۸ متوسط
لیست [2, 7, 4, 9, 3, 6, 8, 5] رو پیمایش کن و یه لیست جدید از اعداد زوج بساز.
مشاهده جواب
nums = [2, 7, 4, 9, 3, 6, 8, 5]
evens = []
for num in nums:
    if num % 2 == 0:
        evens.append(num)
print("اعداد زوج:", evens)
تمرین ۹ متوسط
از کاربر ۵ اسم بگیر و توی لیست ذخیره کن. بعد اسم "علی" رو توی لیست جستجو کن و بگو هست یا نه.
مشاهده جواب
names = []
for i in range(5):
    name = input(f"اسم {i+1}: ")
    names.append(name)

if "علی" in names:
    print("علی توی لیست هست")
else:
    print("علی توی لیست نیست")
تمرین ۱۰ سخت
لیست [1, 2, 3, 2, 4, 2, 5, 2] رو پیمایش کن و تعداد تکرار عدد ۲ رو بشمر.
مشاهده جواب
nums = [1, 2, 3, 2, 4, 2, 5, 2]
count = 0
for num in nums:
    if num == 2:
        count += 1
print("تعداد تکرار 2:", count)
تمرین ۱۱ سخت
دو لیست [1, 2, 3] و [4, 5, 6] رو با هم ترکیب کن (concatenate) و یه لیست جدید بساز.
مشاهده جواب
list1 = [1, 2, 3]
list2 = [4, 5, 6]
combined = list1 + list2
print("ترکیب:", combined)
تمرین ۱۲ سخت
یه لیست از اعداد ۱ تا ۲۰ بساز. بعد با حلقه، اعداد بخش‌پذیر بر ۳ رو توی لیست جدیدی بریز.
مشاهده جواب
nums = list(range(1, 21))
divisible_by_3 = []
for num in nums:
    if num % 3 == 0:
        divisible_by_3.append(num)
print("بخش‌پذیر بر 3:", divisible_by_3)
تمرین ۱۳ سخت
برنامه‌ای بنویس که ۱۰ عدد از کاربر بگیره و توی لیست ذخیره کنه. بعد اعداد رو از کوچیک به بزرگ مرتب کنه و چاپ کنه.
مشاهده جواب
numbers = []
for i in range(10):
    num = int(input(f"عدد {i+1}: "))
    numbers.append(num)
numbers.sort()
print("اعداد مرتب شده:", numbers)
تمرین ۱۴ سخت
لیست ["python", "java", "c++", "javascript"] رو پیمایش کن و طول هر کلمه رو چاپ کن.
مشاهده جواب
languages = ["python", "java", "c++", "javascript"]
for lang in languages:
    print(f"{lang}: {len(lang)} کاراکتر")
✨ پروژه نهایی لیست‌ها پروژه
یه برنامه مدیریت مخاطبین (Contacts) بنویس که:
• یه لیست خالی برای ذخیره اسم‌ها داره
• یه لیست خالی برای ذخیره شماره تلفن‌ها داره
• یه منو با این گزینه‌ها:
  1. اضافه کردن مخاطب (اسم و شماره)
  2. نمایش همه مخاطبین
  3. جستجوی مخاطب (با اسم)
  4. حذف مخاطب (با اسم)
  5. خروج
• برنامه تا وقتی کاربر گزینه 5 رو انتخاب نکنه باید ادامه پیدا کنه.
مشاهده جواب
names = []
phones = []
choice = 0

while choice != 5:
    print("\n" + "="*40)
    print("مدیریت مخاطبین")
    print("1. اضافه کردن مخاطب")
    print("2. نمایش همه مخاطبین")
    print("3. جستجوی مخاطب")
    print("4. حذف مخاطب")
    print("5. خروج")
    choice = int(input("انتخاب کن: "))

    if choice == 1:
        name = input("اسم: ")
        phone = input("شماره: ")
        names.append(name)
        phones.append(phone)
        print("مخاطب اضافه شد!")

    elif choice == 2:
        if len(names) == 0:
            print("لیست خالی است")
        else:
            for i in range(len(names)):
                print(f"{i+1}. {names[i]} - {phones[i]}")

    elif choice == 3:
        search = input("اسم مورد نظر: ")
        found = False
        for i in range(len(names)):
            if names[i] == search:
                print(f"پیدا شد: {names[i]} - {phones[i]}")
                found = True
        if not found:
            print("پیدا نشد")

    elif choice == 4:
        search = input("اسم برای حذف: ")
        found = False
        for i in range(len(names)):
            if names[i] == search:
                del names[i]
                del phones[i]
                print("حذف شد")
                found = True
                break
        if not found:
            print("پیدا نشد")

    elif choice == 5:
        print("خدانگهدار!")
    else:
        print("انتخاب نامعتبر")

نکات مهم لیست‌ها

  • ایندکس لیست از ۰ شروع میشه
  • برای کپی کردن از copy() استفاده کن، نه =
  • موقع پیمایش لیست، اون رو تغییر نده (مثلاً حذف نکن)
  • از enumerate برای دسترسی همزمان به ایندکس و مقدار استفاده کن
  • برش (slicing) خیلی کاربردی‌ست
